1. Release WiiuseJ
	OLD=0.14-SNAPSHOT
	CURRENT=0.14

	cd wiiusej
	find -name pom.xml | xargs sed -i -e 's/$OLD/$CURRENT/g'
	git diff
	mvn clean package

	# Will go to master since there were no other branches up to this point
	git commit -a -m "Releasing $CURRENT"
	git push origin master

	# Commands below can be replaced with creating tag directly on GitHub

	NEW=0.15-SNAPSHOT

	git checkout -b $NEW

	git commit -a -m "Starting $NEW"
	git push origin $NEW
	cd ..
